CREATE TABLESPACE BBSDB

// 유저 정보 생성				
// USER → 계정이름 | IDENTIDIED → 비밀번호 | TABLESPACE → 작업 공간				
// TEMPORARY TABLESPACE → 임시 작업 공간

TEMPORARY TABLESPACE TEMP;				
				
// 접속 및 자원 사용 및 관리자 권한 설정				
GRANT CONNECT, RESOURCE, DBA TO webuser;
